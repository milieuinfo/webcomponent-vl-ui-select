Select dynamische data

    
      <select id="vl-select-dynamisch"
        is="vl-select"
        data-vl-select-deletable>
        <option placeholder>Selecteer item</option>
      </select>

      <button onclick="laadDynamisch()" type="button">Activeer dynamische data</button>

      <script type="text/javascript">
        function laadDynamisch() {
          document.getElementById('vl-select-dynamisch').dress({
            callbackFn: (select) => {
              return [
                {value: 'One', label: 'Label One', disabled: true},
                {value: 'Two', label: 'Label Two', selected: true},
                {value: 'Three', label: 'Label Three'},
              ];
            }
          });
        }
      </script>

Select methoden

    <script type="text/javascript">
      const select = () => {
        return document.getElementById('vl-select-methoden');
      };

      const methodSelectDress = () => {
        select().dress();
      };

      const methodSelectUndress = () => {
        select().undress();
      };

      const methodSelectEnable = () => {
        select().enable();
      };

      const methodSelectDisable = () => {
        select().disable();
      };

      const methodSelectSetValueByChoice = () => {
        select().setValueByChoice('Duitsland');
      };

      const methodSelectRemoveActive = () => {
        select().removeActive();
      };

      const methodSelectPlaceholder = () => {
        select().setValueByChoice('');
      };
    </script>

    <select id="vl-select-methoden"
      is="vl-select"
      data-vl-select-deletable>
      <option placeholder value="">Kies een land</option>
      <option value="België">België</option>
      <option value="Duitsland">Duitsland</option>
      <option value="Frankrijk">Frankrijk</option>
      <option value="Nederland">Nederland</option>
      <option value="Italië">Italië</option>
    </select>

    <button onclick="methodSelectDress()" type="button">dress()</button>
    <button onclick="methodSelectUndress()" type="button">undress()</button>
    <button onclick="methodSelectEnable()" type="button">enable()</button>
    <button onclick="methodSelectDisable()" type="button">disable()</button>
    <button onclick="methodSelectSetValueByChoice()" type="button">setValueByChoice('Duitsland')</button>
